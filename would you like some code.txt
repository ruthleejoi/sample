        async function initAudio() {
            if (!audioCtx) audioCtx = new (window.AudioContext || window.webkitAudioContext)();
            if (audioCtx.state === 'suspended') await audioCtx.resume();
        }

        // Voice synthesis function for menu items
        function speakText(text) {
            if (!window.speechSynthesis) return;
            // Stop any current speech
            window.speechSynthesis.cancel();
            const utterance = new SpeechSynthesisUtterance(text);
            utterance.lang = 'en-US';
            utterance.rate = 0.9; // Slightly slower for clarity
            window.speechSynthesis.speak(utterance);
        }

        function playSFX(type) {
            if (!audioCtx) return;
            initAudio();
            const osc = audioCtx.createOscillator();
            const gain = audioCtx.createGain();
            osc.connect(gain); gain.connect(audioCtx.destination);
            const now = audioCtx.currentTime;
            if (type === 'correct') {
                osc.type = 'triangle'; osc.frequency.setValueAtTime(523.25, now); osc.frequency.exponentialRampToValueAtTime(1046.50, now + 0.1);
                gain.gain.setValueAtTime(0.1, now); gain.gain.exponentialRampToValueAtTime(0.01, now + 0.2);
                osc.start(); osc.stop(now + 0.2);
            } else if (type === 'wrong') {
                osc.type = 'sawtooth'; osc.frequency.setValueAtTime(150, now); osc.frequency.linearRampToValueAtTime(50, now + 0.3);
                gain.gain.setValueAtTime(0.1, now); gain.gain.linearRampToValueAtTime(0.01, now + 0.3);
                osc.start(); osc.stop(now + 0.3);
            } else if (type === 'eat') {
                osc.type = 'square'; osc.frequency.setValueAtTime(600, now); osc.frequency.exponentialRampToValueAtTime(300, now + 0.1);
                gain.gain.setValueAtTime(0.1, now); gain.gain.exponentialRampToValueAtTime(0.01, now + 0.1);
                osc.start(); osc.stop(now + 0.1);
            }
        }

        function startBGM() {
            if (!audioCtx) return;
            const length = 4.0; const sampleRate = audioCtx.sampleRate;
            const buffer = audioCtx.createBuffer(1, sampleRate * length, sampleRate);
            const data = buffer.getChannelData(0);
            for (let i = 0; i < buffer.length; i++) {
                const t = i / sampleRate; const beat = t % 0.5; const bar = t % 4.0;
                const kick = Math.exp(-10 * beat) * Math.sin(2 * Math.PI * 40 * Math.exp(-2 * beat));
                let freq = 110.00; 
                if (bar > 1.0) freq = 123.47; 
                if (bar > 2.0) freq = 98.00; 
                if (bar > 3.0) freq = 110.00; 
                const bass = Math.sin(2 * Math.PI * freq * t) * (beat < 0.2 ? 0.05 : 0);
                const leadT = (t * 8) % 1; const note = freq * 4;
                const lead = (Math.random() * 2 - 1) * Math.exp(-5 * leadT) * 0.01;
                data[i] = (kick * 0.1) + bass + lead;
            }
            bgmNode = audioCtx.createBufferSource(); bgmNode.buffer = buffer; bgmNode.loop = true;
            bgmNode.connect(audioCtx.destination); bgmNode.start();
        }

        function preloadAssets(callback) {
            let loaded = 0; const items = [...foods, ...mealTypes];
            items.forEach(item => {
                const img = new Image();
                img.crossOrigin = "anonymous";
                img.onload = () => { loaded++; if (loaded === items.length) callback(); };
                img.src = item.img;
                assets[item.id] = img;
            });
        }

        function initVocab() {
            const foodList = document.getElementById('food-vocab-list');
            const mealList = document.getElementById('meal-vocab-list');
            
            mealTypes.forEach(m => {
                const container = document.createElement('div');
                container.className = "flex flex-col items-center gap-2";
                const pill = document.createElement('div');
                pill.className = "pill-header w-full";
                pill.innerText = m.label.toUpperCase();
                const card = createCard(m);
                card.querySelector('p').remove();
                const labelContainer = document.createElement('div');
                labelContainer.className = "p-3 text-center bg-zinc-900";
                labelContainer.innerHTML = `<p class="text-white font-bold text-sm leading-tight">${m.label}</p><p class="text-neon-green font-medium text-[10px] my-0.5">${m.phonetic}</p><p class="text-zinc-500 font-bold text-[9px] uppercase">${m.desc}</p>`;
                card.appendChild(labelContainer);
                container.appendChild(pill);
                container.appendChild(card);
                mealList.appendChild(container);
            });

            foods.forEach(f => {
                const card = createCard(f);
                card.querySelector('p').remove();
                const labelContainer = document.createElement('div');
                labelContainer.className = "p-2 text-center bg-zinc-900";
                labelContainer.innerHTML = `<p class="text-white font-bold text-xs leading-tight">${f.label}</p><p class="text-neon-green font-medium text-[9px] mt-0.5">${f.phonetic}</p>`;
                card.appendChild(labelContainer);
                foodList.appendChild(card);
            });
        }

        function createCard(item) {
            const card = document.createElement('div');
            card.className = "vocab-card w-full";
            card.innerHTML = `<img src="${item.img}" class="card-img"><p class="p-2 text-white font-bold uppercase text-[10px]">${item.label}</p>`;
            card.onclick = (e) => {
                e.stopPropagation();
                // Speak the item label when clicked
                speakText(item.label);
                zoomImageWithText(item.img, item.label, `${item.phonetic}\n${item.article ? `${item.article} ${item.label}` : item.desc}`);
            };
            return card;
        }

        function zoomImageWithText(src, title, desc) {
            const z = document.getElementById('full-screen-zoom');
            if (src) {
                document.getElementById('zoom-img').style.display = 'block';
                document.getElementById('zoom-img').src = src;
            } else {
                document.getElementById('zoom-img').style.display = 'none';
            }
            document.getElementById('zoom-title').innerText = title;
            document.getElementById('zoom-desc').innerText = desc;
            z.style.display = 'flex';
        }

        function zoomRule(title, desc) { 
            zoomImageWithText(null, title, desc); 
        }
        function closeZoom() { 
            document.getElementById('full-screen-zoom').style.display = 'none'; 
            if (window.speechSynthesis) window.speechSynthesis.cancel();
        }
        function showGrammarPreview() { document.getElementById('preview-vocab').style.display = 'none'; document.getElementById('preview-grammar').style.display = 'flex'; }
        function showBilingualInstructions() { document.getElementById('preview-grammar').style.display = 'none'; document.getElementById('preview-instructions').style.display = 'flex'; }

        async function startGame() {
            await initAudio();
            document.getElementById('preview-instructions').style.display = 'none';
            resetGame(true);
        }

        function resetGame(fullReset) {
            if (fullReset) { score = 0; lives = 3; speedBoost = 0; currentMeal = 'breakfast'; }
            const maxX = Math.floor(canvas.width/gridSize);
            const maxY = Math.floor(canvas.height/gridSize);
            snake = [{x: Math.floor(maxX/2), y: Math.floor(maxY/2)}];
            dx = 0; dy = 0;
            gameActive = true; 
            isPaused = false; 
            isWaitingForStart = true; 
            updateUI();
            spawnFood();
            document.getElementById('start-prompt-overlay').style.display = 'flex';
            requestAnimationFrame(gameLoop);
        }

        function handleDirection(dir) {
            if (!isWaitingForStart && !gameActive) return;
            
            let newDx = 0, newDy = 0;
            if (dir === 'up' && dy === 0) { newDx = 0; newDy = -1; }
            else if (dir === 'down' && dy === 0) { newDx = 0; newDy = 1; }
            else if (dir === 'left' && dx === 0) { newDx = -1; newDy = 0; }
            else if (dir === 'right' && dx === 0) { newDx = 1; newDy = 0; }
            else { return; } // Invalid direction (e.g. going backward)

            dx = newDx; dy = newDy;

            if (isWaitingForStart) {
                document.getElementById('start-prompt-overlay').style.display = 'none';
                isWaitingForStart = false;
            }
        }

        function respawn() {
            score = Math.max(0, score - 5);
            document.getElementById('gameover-overlay').style.display = 'none';
            resetGame(false);
        }

        function showFinalResults() {
            document.getElementById('death-title').innerText = "MISSION CLEAR";
            document.getElementById('death-title').className = "text-4xl font-black text-neon-green mb-2 italic";
            document.getElementById('life-remaining-ui').classList.add('hidden');
            document.getElementById('final-results').classList.remove('hidden');
            document.getElementById('restart-btn').classList.remove('hidden');
            document.getElementById('final-score-display').innerText = score;
        }

        function updateUI() {
            document.getElementById('score').innerText = score.toString().padStart(3, '0');
            document.getElementById('meal-indicator').innerText = meals[currentMeal].label;
            document.getElementById('time-indicator').innerText = meals[currentMeal].time;
            document.getElementById('lives-container').innerHTML = '<span class="text-red-500 text-2xl mx-0.5">♥</span>'.repeat(lives);
            updateClock();
        }

        function updateClock() {
            const mealData = meals[currentMeal];
            const hourDeg = (mealData.hour % 12) * 30 + (mealData.min / 60) * 30;
            const minDeg = mealData.min * 6;
            document.getElementById('hour-hand').style.transform = `translateX(-50%) rotate(${hourDeg}deg)`;
            document.getElementById('min-hand').style.transform = `translateX(-50%) rotate(${minDeg}deg)`;
        }

        function spawnFood() {
            const maxX = Math.floor(canvas.width / gridSize) - 1;
            const maxY = Math.floor(canvas.height / gridSize) - 1;
            let newX, newY;
            do {
                newX = Math.floor(Math.random() * maxX);
                newY = Math.floor(Math.random() * maxY);
            } while (snake.some(s => s.x === newX && s.y === newY));
            food.x = newX; food.y = newY;
            food.type = foods[Math.floor(Math.random() * foods.length)].id;
        }

        window.addEventListener('keydown', e => {
            if (!gameActive || isPaused) return;
            const dirMap = { ArrowUp: 'up', ArrowDown: 'down', ArrowLeft: 'left', ArrowRight: 'right' };
            if (dirMap[e.key]) handleDirection(dirMap[e.key]);
        });

        function gameLoop(t) {
            if (!gameActive || isPaused) return;
            if (isWaitingForStart) { draw(); requestAnimationFrame(gameLoop); return; }
            
            const currentSpeed = Math.max(70, baseSpeed - speedBoost);
            if (t - lastTimestamp > currentSpeed) {
                move(); draw(); lastTimestamp = t;
            }
            requestAnimationFrame(gameLoop);
        }

        function move() {
            if (dx === 0 && dy === 0) return;
            const head = {x: snake[0].x + dx, y: snake[0].y + dy};
            const maxX = Math.floor(canvas.width/gridSize);
            const maxY = Math.floor(canvas.height/gridSize);
            if (head.x < 0 || head.x >= maxX || head.y < 0 || head.y >= maxY || snake.some(s => s.x === head.x && s.y === head.y)) {
                lives--; updateUI(); playSFX('wrong'); gameActive = false;
                document.getElementById('gameover-overlay').style.display = 'flex';
                document.getElementById('remaining-lives-text').innerText = lives;
                if (lives <= 0) showFinalResults();
                return;
            }
            snake.unshift(head);
            if (head.x === food.x && head.y === food.y) { playSFX('eat'); triggerPrompt(); }
            else if (pendingGrowth > 0) { pendingGrowth--; }
            else { snake.pop(); }
        }

        function draw() {
            ctx.fillStyle = '#000'; ctx.fillRect(0, 0, canvas.width, canvas.height);
            ctx.strokeStyle = '#222'; ctx.lineWidth = 1;
            for(let i=0; i<canvas.width; i+=gridSize) { ctx.beginPath(); ctx.moveTo(i,0); ctx.lineTo(i,canvas.height); ctx.stroke(); }
            for(let i=0; i<canvas.height; i+=gridSize) { ctx.beginPath(); ctx.moveTo(0,i); ctx.lineTo(canvas.width,i); ctx.stroke(); }
            
            const img = assets[food.type];
            if (img) {
                ctx.save(); ctx.shadowBlur = 15; ctx.shadowColor = "white";
                ctx.drawImage(img, food.x * gridSize + 2, food.y * gridSize + 2, gridSize - 4, gridSize - 4);
                ctx.restore();
            }

            snake.forEach((s, i) => {
                ctx.fillStyle = i === 0 ? '#39FF14' : '#1a7a0a';
                ctx.shadowBlur = i === 0 ? 10 : 0; ctx.shadowColor = '#39FF14';
                ctx.fillRect(s.x*gridSize + 2, s.y*gridSize + 2, gridSize - 4, gridSize - 4);
                if (i === 0) {
                    ctx.fillStyle = 'black';
                    if (dx === 1 || (dx === 0 && dy === 0)) {
                        ctx.fillRect(s.x*gridSize+25, s.y*gridSize+10, 5, 5);
                        ctx.fillRect(s.x*gridSize+25, s.y*gridSize+25, 5, 5);
                    } else if (dx === -1) {
                        ctx.fillRect(s.x*gridSize+10, s.y*gridSize+10, 5, 5);
                        ctx.fillRect(s.x*gridSize+10, s.y*gridSize+25, 5, 5);
                    } else if (dy === -1) {
                        ctx.fillRect(s.x*gridSize+10, s.y*gridSize+10, 5, 5);
                        ctx.fillRect(s.x*gridSize+25, s.y*gridSize+10, 5, 5);
                    } else if (dy === 1) {
                        ctx.fillRect(s.x*gridSize+10, s.y*gridSize+25, 5, 5);
                        ctx.fillRect(s.x*gridSize+25, s.y*gridSize+25, 5, 5);
                    }
                }
            });
        }

        function triggerPrompt() {
            isPaused = true;
            const foodData = foods.find(f => f.id === food.type);
            const otherFood = foods.filter(f => f.id !== food.type)[Math.floor(Math.random() * (foods.length-1))];
            const usePattern1 = Math.random() > 0.5;
            let questionData;

            if (usePattern1) {
                questionData = {
                    q: `Would you like some ${foodData.label}?`,
                    opts: [
                        { text: "Yes, please.", correct: true }, 
                        { text: `No, thank you. I'd like some ${otherFood.label}.`, correct: true }, 
                        { text: "I'd like some soup.", correct: false }
                    ]
                };
            } else {
                questionData = {
                    q: `What would you like for ${currentMeal}?`,
                    opts: [
                        { text: `I'd like some ${foodData.label}.`, correct: true }, 
                        { text: `Yes, please.`, correct: false },
                        { text: `No, thank you.`, correct: false }
                    ]
                };
            }

            const modal = document.getElementById('grammar-modal');
            const optCont = document.getElementById('options-container');
            const feedback = document.getElementById('feedback');
            document.getElementById('prompt-img').src = foodData.img;
            document.getElementById('prompt-text').innerText = questionData.q;
            optCont.innerHTML = ""; feedback.classList.add('hidden');
            
            questionData.opts.sort(() => Math.random() - 0.5).forEach((optObj) => {
                const b = document.createElement('button');
                b.className = "w-full p-5 rounded-xl border border-zinc-800 bg-zinc-900 hover:bg-zinc-800 font-bold transition-all text-white text-lg text-left";
                b.innerText = optObj.text;
                b.onclick = () => {
                    if (optObj.correct) {
                        playSFX('correct'); score += 5; speedBoost += 5; pendingGrowth += 2;
                        document.getElementById('feedback-text').innerText = "CORRECT! +5 PTS";
                        document.getElementById('feedback-text').className = "text-neon-green font-bold text-xl mb-4";
                    } else {
                        playSFX('wrong'); score = Math.max(0, score - 2);
                        document.getElementById('feedback-text').innerText = "WRONG! -2 PTS";
                        document.getElementById('feedback-text').className = "text-red-500 font-bold text-xl mb-4";
                    }
                    optCont.classList.add('pointer-events-none', 'opacity-30');
                    feedback.classList.remove('hidden');
                    currentMeal = meals[currentMeal].next; updateUI();
                };
                optCont.appendChild(b);
            });
            modal.style.display = "flex";
        }

        function closePrompt() {
            document.getElementById('grammar-modal').style.display = "none";
            document.getElementById('options-container').classList.remove('pointer-events-none', 'opacity-30');
            isPaused = false; 
            isWaitingForStart = true;
            document.getElementById('start-prompt-overlay').style.display = 'flex';
            spawnFood(); 
            requestAnimationFrame(gameLoop);
        }

        window.onload = () => {
            preloadAssets(() => {
                initVocab();
                document.getElementById('preview-vocab').style.display = 'flex';
            });
        };
